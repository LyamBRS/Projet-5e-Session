116void processusDetection_lit(void)
117{
118 if(ModuleData.Mode == Modes.operation)
119 {
120 if (interfaceVL6180x_litUneDistance(&distance) < 0)
121 {
122 printf("erreur: interfaceVL6180x_litUneDistance");
123 fflush(stdout);
124 return;
125 }
126 if(dernieredistance <= 19)
127 {
128 if(distance <= 19)
129 {
130 //On a vu deux fois la meme distance donc c'est probablement le cube
131 printf("CUUUBE\n");
132 fflush(stdout);
133 processusDetection.information = PROCESSUSDETECTION_INFORMATION_CUBEDETECTED;
134 serviceBaseDeTemps_execute[PROCESSUSDETECTION_PHASE] = processusDetection_alignement;
135 return;
136 }
137 }
138
139 dernieredistance = distance;
140 printf("%2.1f\n", distance);
141 fflush(stdout);
142 serviceBaseDeTemps_execute[PROCESSUSDETECTION_PHASE] = processusDetection_Bouge;
143 }
144 else
145 {
146 if(ModuleData.Mode == Modes.pause)
147 {
148 ModuleData.State = States.paused;
149 }
150 }
