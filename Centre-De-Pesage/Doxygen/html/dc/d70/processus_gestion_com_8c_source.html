79 if(processusPesage.unite == PROCESSUSPESAGE_UNITE_G)

81 ModuleData_SetUnits(Values.unit_Metric);

83 else
84 {
85 ModuleData_SetUnits(Values.unit_Imperial);

90 //Print connection successful
